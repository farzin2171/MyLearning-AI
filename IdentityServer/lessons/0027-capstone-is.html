Capstone — Build WorkSpace IS

~35 minutes Hands-on build Prerequisites: Lesson 26 design, .NET 8 SDK, Duende IS templates

Design complete. Scaffold starts now. By the end of this lesson you will have a running Duende IS instance at https://localhost:5001 that issues tokens for the WorkSpace system — discoverable, loginable, and enriching tokens with tenant claims.

Build order Follow the steps in sequence. Each builds on the previous. Run the verify check after each step before moving on.

using Duende.IdentityServer.Models;

public static class Config
{
    public static IEnumerable<IdentityResource> IdentityResources => new[]
    {
        new IdentityResources.OpenId(),
        new IdentityResources.Profile(),
        new IdentityResources.Email(),
    };

    public static IEnumerable<ApiScope> ApiScopes => new[]
    {
        new ApiScope("calendar.read",
                     displayName: "Read calendar events",
                     userClaims:  new[] { "tenant_id", "role" }),
        new ApiScope("calendar.write",
                     displayName: "Write calendar events",
                     userClaims:  new[] { "tenant_id", "role" }),
        new ApiScope("contacts.read",
                     displayName: "Read contacts",
                     userClaims:  new[] { "tenant_id" }),
        new ApiScope("contacts.write",
                     displayName: "Write contacts",
                     userClaims:  new[] { "tenant_id" }),
    };

    public static IEnumerable<Client> Clients => new[]
    {
        // Interactive web client
        new Client
        {
            ClientId              = "workspace-web",
            ClientName            = "WorkSpace Web App",
            AllowedGrantTypes     = GrantTypes.Code,
            RequirePkce           = true,
            RequireClientSecret   = false,
            RedirectUris          = { "https://localhost:5002/signin-oidc" },
            PostLogoutRedirectUris = { "https://localhost:5002/signout-callback-oidc" },
            AllowOfflineAccess    = true,
            AllowedScopes         =
            {
                "openid", "profile", "email",
                "calendar.read", "calendar.write",
                "contacts.read"
            }
        },

        // Background reporting worker
        new Client
        {
            ClientId          = "reports-worker",
            ClientName        = "WorkSpace Reports Worker",
            AllowedGrantTypes = GrantTypes.ClientCredentials,
            ClientSecrets     = { new Secret("worker-secret".Sha256()) },
            AllowedScopes     = { "calendar.read", "contacts.read" }
        }
    };
}

4

Implement IProfileService

Create Services/WorkSpaceProfileService.cs. This reads the session's sub, finds the user in TestUsers, and adds their tenant_id and role when those claim types are requested.

Services/WorkSpaceProfileService.cs
using Duende.IdentityServer.Extensions;
using Duende.IdentityServer.Models;
using Duende.IdentityServer.Services;

public class WorkSpaceProfileService : IProfileService
{
    public Task GetProfileDataAsync(ProfileDataRequestContext context)
    {
        var sub  = context.Subject.GetSubjectId();
        var user = TestUsers.Users.FirstOrDefault(u => u.SubjectId == sub);

        if (user == null) return Task.CompletedTask;

        var claims = new List<Claim>();

        if (context.RequestedClaimTypes.Contains("tenant_id"))
        {
            var tenantClaim = user.Claims.FirstOrDefault(c => c.Type == "tenant_id");
            if (tenantClaim != null) claims.Add(tenantClaim);
        }

        if (context.RequestedClaimTypes.Contains("role"))
        {
            var roleClaim = user.Claims.FirstOrDefault(c => c.Type == "role");
            if (roleClaim != null) claims.Add(roleClaim);
        }

        if (context.RequestedClaimTypes.Contains("email"))
        {
            var emailClaim = user.Claims.FirstOrDefault(c => c.Type == "email");
            if (emailClaim != null) claims.Add(emailClaim);
        }

        context.IssuedClaims.AddRange(claims);
        return Task.CompletedTask;
    }

    public Task IsActiveAsync(IsActiveContext context)
    {
        var sub  = context.Subject.GetSubjectId();
        context.IsActive = TestUsers.Users.Any(u => u.SubjectId == sub);
        return Task.CompletedTask;
    }
}
5

Wire everything in Program.cs

Program.cs
var builder = WebApplication.CreateBuilder(args);

builder.Services.AddRazorPages();

builder.Services
    .AddIdentityServer(options =>
    {
        options.Events.RaiseErrorEvents       = true;
        options.Events.RaiseInformationEvents = true;
        options.Events.RaiseFailureEvents     = true;
        options.Events.RaiseSuccessEvents     = true;
    })
    .AddInMemoryIdentityResources(Config.IdentityResources)
    .AddInMemoryApiScopes(Config.ApiScopes)
    .AddInMemoryClients(Config.Clients)
    .AddTestUsers(TestUsers.Users)          // wires the QuickStart user store
    .AddProfileService<WorkSpaceProfileService>()
    .AddDeveloperSigningCredential();

var app = builder.Build();

app.UseStaticFiles();
app.UseRouting();
app.UseIdentityServer();    // includes UseAuthentication
app.UseAuthorization();
app.MapRazorPages().RequireAuthorization();

app.Run();
Verify → dotnet build must succeed with zero errors before the next step.

Three Verification Checks

With IS running, open these URLs in a browser:

Check 1 — Discovery document https://localhost:5001/.well-known/openid-configuration
You should see JSON with issuer, authorization_endpoint, token_endpoint, jwks_uri.
Check 2 — JWKS https://localhost:5001/.well-known/openid-configuration/jwks
You should see the public key JSON. This is what the Calendar API will fetch to validate token signatures.
Check 3 — Login page https://localhost:5001/Account/Login
You should see the QuickStart login form. Log in as alice / alice. After login, the diagnostics page at /Diagnostics shows the active session claims.

Knowledge Check

1. The .AddTestUsers(TestUsers.Users) call in Program.cs does two things. Which pair is correct?

2. Alice logs in and requests scope=calendar.read. Which claim will appear in her access token that would not appear if Bob (a different tenant) requested the same scope?

3. Why does WorkSpaceProfileService filter by context.RequestedClaimTypes before adding claims?
